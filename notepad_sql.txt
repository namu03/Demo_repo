create table Member
(
Member_id number(5) primary key,
Member_Name varchar2(30) not null,
member_address varchar2(50) not null,
Acc_Open_Date date not null,
Membership_type varchar2(20) not null,
fees_paid number(4),
Max_Books_Allowed number(2),
penalty_amt number(7,2)
);

create table Books
(
Book_no number(6) primary key,
Book_name varchar2(30) not null,
Author_name varchar2(30) not null,
cose number(7,2) not null,
category char(10) not null
);

create table issue
(
Lib_Issue_Id number(10) primary key,
Book_no number(6) references Books(Book_no) on delete cascade,
Member_Id number(5) references Member(Member_Id) on delete cascade,
Issue_date Date not null,
return_date date
);
